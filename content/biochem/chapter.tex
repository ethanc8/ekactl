\section{pH and buffers}

\subsection{pH and pOH}

$[\ce{H+}]$ is the molarity of \ce{H+} in whatever solution we're measuring the pH of.

\begin{align*}
  \pH &= - \log_{10} [\ce{H+}] & [\ce{H+}] = 10^{-\pH} \\
  \pOH &= - \log_{10} [\ce{OH-}] & [\ce{OH-}] = 10^{-\pOH} \\
  \pH + \pOH &= 14
\end{align*}

\subsection{Strong acids and bases}

The following are the strong acids to memorize:
\begin{itemize}
  \item \ce{H2SO4} (sulfuric acid)
  \item \ce{HNO3} (nitric acid)
  \item \ce{HCl} (hydrochloric acid)
\end{itemize}

The following are the strong bases to memorize:
\begin{itemize}
  \item \ce{NaOH} (sodium hydroxide)
  \item \ce{KOH} (potassium hydroxide)
  \item \ce{Ca(OH)2} (calcium hydroxide)
\end{itemize}

\begin{definition}[Dissociation of strong acids and bases]
  The reaction in which a strong acid or base breaks apart into a proton (\ce{H+}) (for a strong acid) or a hydronium (\ce{OH-}) (for a strong base) and another ion.

  Strong acids and bases dissociate completely in water.
\end{definition}


\subsection{Weak acids}

\begin{definition}[Conjugate base]
  The non-proton product of an acid ionization reaction in which an acid loses its proton.
\end{definition}

\begin{definition}[Ionization of a weak acid]
  The reaction in which part of a weak acid breaks apart into a proton and its conjugate base. The conjugate base is often written as \ce{A-}.
  \[
    \ce{HA(aq) <=> H+(aq) + A-(aq)}
  \]

  This often occurs in water. Note that only a small amount of most weak acids ionize.
\end{definition}


\begin{definition}[Equilibrium constant for ionization of a weak acid]
  \[
    K_a = \frac{[\ce{H+}][\ce{A-}]}{[\ce{HA}]}
  \]
  The negative log of $K_a$ is called $\pKa$:
  \[
    \pKa = - \log_{10} K_a
  \]
\end{definition}

\begin{theorem}[Henderson-Hasselbalch equation]
  \[
    \pH = \pKa + \log \left(\frac{[\ce{A-}]}{[\ce{HA}]}\right) = \pKa + \log \left(\frac{[\text{conjugate base}]}{[\text{acid}]}\right)
  \]
\end{theorem}

\subsection{Buffers}

\begin{definition}[Buffer]
  A solution that can resist change in pH upon the addition of an acid or a base.
\end{definition}

Buffers can be created using a weak acid and its conjugate base, or a weak base and its conjugate acid.
